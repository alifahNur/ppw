# Temukan semua artikel dengan kelas 'text-card'
articles = soup.find_all('article', class_='text-card')

# Inisialisasi counter untuk artikel
article_counter = 1

# Iterasi melalui setiap artikel
for article in articles:
    # Ambil judul dan link
    title = article.find('h2', class_='title')
    if title:
        title_text = title.text.strip()
        link = title.find('a')['href']
        print(f"{article_counter}. Judul: {title_text}")
        print(f"   Link: {link}")

    # Ambil isi artikel
    content = article.find('p')
    if content:
        print(f"   Isi: {content.text.strip()}")

    # Ambil tanggal publikasi
    date = article.find('h4', class_='date')
    if date:
        date_text = date.text.strip()

        # Konversi "X jam lalu" ke waktu spesifik
        if "jam lalu" in date_text:
            hours_ago = int(date_text.split(' ')[0])
            publish_time = datetime.now() - timedelta(hours=hours_ago)
            print(f"   Tanggal: {publish_time.strftime('%d-%m-%Y %H:%M')}")
        else:
            print(f"   Tanggal: {date_text}")

    # Increment counter
    article_counter += 1
