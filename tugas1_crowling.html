# Mengambil semua artikel dari halaman
articles = soup.find_all('article')  # Ubah selector menjadi 'article' jika setiap artikel dibungkus dalam elemen <article>

# Set untuk menyimpan href yang sudah dilihat agar tidak terjadi duplikasi
seen_links = set()

# Counter untuk artikel yang diproses
link_counter = 0

# Iterasi melalui setiap artikel
for article in articles:
    # Ambil link artikel
    link = article.find('a', class_='ui--a articles--iridescent-list--text-item__title-link')
    if link:
        href = link.get('href')
        title = link.get('title').strip()

        if href and href not in seen_links:
            print(f"Artikel {link_counter + 1}:")
            print(f"   Judul: {title}")
            print(f"   Link: {href}")

            # Ambil tanggal publikasi
            date = article.find('time', class_='articles--iridescent-list--text-item__time')
            if date:
                date_text = date.text.strip()
                print(f"   Tanggal: {date_text}")

            # Ambil kategori artikel
            category = article.find('a', class_='articles--iridescent-list--text-item__category')
            if category:
                category_text = category.text.strip()
                print(f"   Kategori: {category_text}")

            # Mencari dan mencetak ringkasan artikel yang sesuai
            summary_div = article.find('div', class_='articles--iridescent-list--text-item__summary articles--iridescent-list--text-item__summary-seo')
            if summary_div:
                summary_text = summary_div.text.strip()
                print(f"   Ringkasan: {summary_text}")

            # Tambahkan href ke set untuk menghindari duplikasi
            seen_links.add(href)
            link_counter += 1
            print()  # Spasi untuk memisahkan artikel

# Menampilkan total artikel yang ditemukan
print(f"Total artikel yang ditampilkan: {link_counter}")

# Kumpulkan semua teks artikel (judul, ringkasan)
articles = soup.find_all('article')
corpus = []  # Korpus untuk menyimpan teks artikel

for article in articles:
    text_content = []  # List untuk menyimpan konten setiap artikel

    # Ambil judul
    link = article.find('a', class_='ui--a articles--iridescent-list--text-item__title-link')
    if link:
        title = link.get('title').strip()
        text_content.append(title)

    # Ambil ringkasan artikel
    summary_div = article.find('div', class_='articles--iridescent-list--text-item__summary articles--iridescent-list--text-item__summary-seo')
    if summary_div:
        summary_text = summary_div.text.strip()
        text_content.append(summary_text)

    # Gabungkan semua teks (judul + ringkasan)
    article_text = ' '.join(text_content)
    corpus.append(article_text)
